emplace_back()和push_back()的区别是emplace_back()做了优化，只调用一次拷贝函数，而push_back()调用两次。

emplace_back()和push_back()的区别具体可以见链接

    1. 两句话中的不常见单词easy
class Solution {
public:
    vector<string> uncommonFromSentences(string s1, string s2) {
        map<string,int>mp;
        string s=s1+" "+s2,tmp;
        vector<string>res;
        stringstream ss(s);
        while(ss>>tmp){
            mp[tmp]++;
        }
        for(auto [str,t]:mp){
            if(t==1)res.push_back(str);
        }
        return res;
    }
};

map记录，由于有空格，使用stringstream流来输入输出(类似于c语言中的sscanf、sprintf函数)，最后遍历map两个字符串中值为1的push到res中。

class Solution {
public:
    Node* copyRandomList(Node* head) {
        if(head==NULL)return head;
        for(Node *node=head,*copy=NULL;node;node=node->next->next){
            copy=new Node(node->val);
            copy->next=node->next;
            node->next=copy;
        }
        for(Node *node=head;node;node=node->next->next){
            if(node->random)node->next->random=node->random->next;
        }
        Node *newhead=head->next;
        for(Node *node=head,*tmp=NULL;node&&node->next;){
            tmp=node->next;
            node->next=tmp->next;
            node=tmp;
        }

        return newhead;
    }
};

class Solution {
public:
    int hIndex(vector<int>& citations) {
        int i=0,j=citations.size();
        while(i<j&&j-i>citations[i])i++;
        return j-i;
    }
};
    1. 平方数之和medium
class Solution {
public:
    bool judgeSquareSum(int c) {
        int max=sqrt(c);
        long long i=0,j=max;
        long long total;
        while(i<=j){
            total=i*i+j*j;
            if(total>c)j--;
            else if(total<c)i++;
            else return true;
        }
        return false;
    }
};
