    1. 根据给定数字划分数组

class Solution {
public:
    vector<int> pivotArray(vector<int>& nums, int pivot) {
        vector<int>low,equal,large;
        for(int i=0;i<nums.size();i++){
            if(nums[i]>pivot)large.push_back(nums[i]);
            else if(nums[i]==pivot)equal.push_back(nums[i]);
            else low.push_back(nums[i]);
        }
        vector<int>res;
        for(int i=0;i<low.size();i++){
            res.push_back(low[i]);
        }
        for(int i=0;i<equal.size();i++){
            res.push_back(equal[i]);
        }
        for(int i=0;i<large.size();i++){
            res.push_back(large[i]);
        }
        return res;
    }
};

这题我是使用了low、equal、large，3个vector容器来存储遍历数组过程中出现的数据，最后根据顺序分别将三个容器中的元素弹出来。

class Solution {
public:
    vector<int> sortEvenOdd(vector<int>& nums) {
        if(nums.size()<=2)return nums;
        vector<int>a,b,res;
        for(int i=0;i<nums.size();i=i+2){
            a.push_back(nums[i]);
        }
        for(int i=1;i<nums.size();i=i+2){
            b.push_back(nums[i]);
        }
        sort(a.begin(),a.end());
        sort(b.begin(),b.end());
        int total=0;
        int p=0,q=b.size()-1;
        
        while(total<nums.size()){
            res.push_back(a[p]);
            total++;
            if(total==nums.size())break;
            res.push_back(b[q]);
            total++;
            p++;
            q--;
        }
        return res;
    }
};

class Solution {
public:
    long long smallestNumber(long long num) {
        int tag=0;
        if(num<0){
            tag=1;
            num=-num;
        }
        else if(num==0)return 0;
        vector<int>st;
        while(num){
            st.push_back(num%10);
            num/=10;
        }
        sort(st.begin(),st.end());
        int start=0;
        while(st[start]==0)start++;
        long long res=0;
        if(tag==0){
            long long p=1;
            for(int i=st.size()-1;i>=start+1;i--){
                res+=st[i]*p;
                p*=10;
            }
            for(int i=0;i<start;i++)p*=10;
            res+=st[start]*p;
            return res;
        }
        else{
            long long q=pow(10,st.size()-1);
            for(int i=st.size()-1;i>=start;i--){
                res+=st[i]*q;
                q/=10;
            }
            return -res;
        }
        }

class Bitset {
public:
    int *bit;

    Bitset(int size) {
        bit=new int[size];
        siz=size;
        memset(bit,0,size*4);
    }
    
    void fix(int idx) {
        bit[idx]=1;

        bit[idx]=0;

    void flip() {
        for(int i=0;i<siz;i++){

    bool all() {

考虑到，除了toString和flip这两个函数其他函数都是O(1)，flip()、toString()这两个函数是O(1)，而题目中限制了toString()函数的使用次数，所以我们应该考虑flip()函数的时间复杂度。 as we all know, 按位取反两次其实就变回去了，所以我们只要记录flip的执行次数，这里我是直接使用一个bool值来记录，最后提交通过！代码如下：

class Bitset {
    int n;
    int sum;
    bool flipped;
    vector<bool> v;
    
public:
    Bitset(int size) {
        n = size;
        sum = 0;
        flipped = false;
        v = vector<bool>(n);
    }
    
    void fix(int idx) {
        if (!(v[idx] ^ flipped))
            v[idx] = !v[idx], sum++;
    }
    
    void unfix(int idx) {
        if (v[idx] ^ flipped)
            v[idx] = !v[idx], sum--;
    }
    
    void flip() {
        flipped = !flipped;
        sum = n - sum;
    }
    
    bool all() {
        return sum == n;
    }
    
    bool one() {
        return sum > 0;
    }
    
    int count() {
        return sum;
    }
    
    string toString() {
        string ans;
        for (int i = 0; i < n; ++i)
            if (v[i] ^ flipped)
                ans.push_back('1');
            else
                ans.push_back('0');
        return ans;
    }
};
