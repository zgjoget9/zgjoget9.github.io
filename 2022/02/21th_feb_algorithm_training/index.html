class Solution {
public:
    string modifyString(string s) {
        if (s[0] == '?')s[0] = 'a';
        for (int i = 1; i < s.size(); i++) {
            if (s[i] == '?') {
                for (char tmp = 'a'; tmp <= 'c'; tmp++) {
                    if (s[i - 1] != tmp) {
                        s[i] = tmp;
                        break;
                    }
                }
            }
            else if (s[i] == s[i - 1]) {
                s[i - 1] = ((s[i - 1]- 'a' + 1) % 26)+'a';
                if(i-2>=0&&s[i-1]==s[i-2])s[i - 1] = ((s[i - 1]- 'a' + 1) % 26)+'a';
            }
            
            else continue;
        }
        return s;
    }
};

    1. 最长快乐字符串medium

优先队列

class Solution {
public:
    string longestDiverseString(int a, int b, int c) {
        priority_queue<pair<int,int>>pq;
        if(a)pq.push({a,0});
        if(b)pq.push({b,1});
        if(c)pq.push({c,2});    
        string ans;
        while(!pq.empty()){
            int sz=ans.size();
            pair<int,int>tmp=pq.top();
            pq.pop();
            if(sz>=2&&tmp.second+'a'==ans[sz-1]&&tmp.second+'a'==ans[sz-2]){
                if(pq.empty())break;
                pair<int,int>change=pq.top();
                pq.pop();
                pq.push(tmp);
                ans+=change.second+'a';
                change.first-=1;
                if(change.first!=0)pq.push(change);
            }
            else{
                ans+='a'+tmp.second;
                tmp.first-=1;
                if(tmp.first!=0)pq.push(tmp);
            }
        }
        return ans;

    }
};

    1. 最长回文子串medium

从头到尾扫一遍字符串，如果当前位置到s的末尾的距离*2小于当前计算的tmp_max的长度就提前结束遍历

(因为后面的字符加一起也没已经算出来的长，所以就不用算了)

思考一下其实回文总共就只有两种情况

  • 以单个字符对称

如：cbabc以中间的a对称

  • 以两个字符对称

如：cbaabc以中间的aa对称

第一种情况只需要一直判断s[location-i]==s[location+i]

第二种情况在遍历字符串的时候顺便判断一下当前字符与前面一个字符是否相同。

相同的话就需要判断s[location-i-1]==s[location+i]到结束

要时刻注意不要越界

在当前location计算结束(即一次循环结束)时将当前location计算出的tmp与目前为止计算出的最长字符串比较长度， 如果当前location计算的长度更长就更新目前为止计算出的最长字符串

class Solution {
public:
    string longestPalindrome(string s) {
        int sz=s.size();
        string tmp_max;
        if(sz==1)return s;
        if(sz==2){
            if(s[0]==s[1])return s;
            else return s.substr(0,1);
        }
        int left=1;
        string tmp;

        while(left<sz){
            if((sz-left)*2<tmp_max.size())break;
            int i=0;
            while(s[left-i]==s[left+i]){
                i++;
                if(left-i<0||left+i>=sz)break;
            }
            i--;
            tmp=s.substr(left-i,2*i+1);
            if(tmp.size()>tmp_max.size())tmp_max=tmp;
            
            if(s[left]==s[left-1]){
                int j=0;
                while(s[left-j-1]==s[left+j]){
                    j++;
                    if(left-j-1<0||left+j>=sz)break;
                }
                j--;
                tmp=s.substr(left-1-j,2*j+2);
                if(tmp.size()>tmp_max.size())tmp_max=tmp;
            }
            left++;
        }

        return tmp_max;
    }

};

class Solution {
public:
    int lengthOfLongestSubstring(string s) {
        int tmp_max=0;
        for(int i=0;i<s.size();i++){
            if(s.size()-i<=tmp_max)break;
            unordered_map<char,int>mp;
            int count=0;
            for(int j=i;j<s.size();j++){
                mp[s[j]]++;
                if(mp[s[j]]>1)break;
                else count++;
            }
            if(count>tmp_max)tmp_max=count;
        }
        return tmp_max;
    }
};

    1. 合并K个升序链表hard
/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode() : val(0), next(nullptr) {}
 *     ListNode(int x) : val(x), next(nullptr) {}
 *     ListNode(int x, ListNode *next) : val(x), next(next) {}
 * };
 */
class Solution {
public:
    ListNode* mergeKLists(vector<ListNode*>& lists) {
        int len = lists.size();
        if(len==0)return NULL;
        ListNode* p[len];
        for(int i=0;i<len;i++){
            p[i]=lists[i];
        }
        ListNode *headnode=new ListNode();
        ListNode *t=headnode;

        while(1){
            int remain=len;
            int min=INT_MAX,location;
            for(int i=0;i<len;i++){

                if(p[i]==NULL){
                    remain--;
                    continue;
                }

                else if(p[i]->val<min){
                    min=p[i]->val;
                    location=i;
                }

            }
            if(remain==0)return headnode->next;
            ListNode *node=new ListNode(p[location]->val);
            p[location]=p[location]->next;
            t->next=node;
            t=t->next;
        }
        return headnode->next;
    }
};
